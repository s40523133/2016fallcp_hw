<script type="text/python">
from browser import document
import math
 
# 準備繪圖畫布
canvas = document["dc1"]
ctx = canvas.getContext("2d")

def circle(x,y,r):
    ctx.beginPath()
    ctx.arc(x, y, r, 0, math.pi*2, True)
    ctx.fill()
    ctx.closePath()

def line(x1,y1,x2,y2):
    ctx.beginPath()
    ctx.lineWidth = 1
    ctx.moveTo(x1, y1)
    ctx.lineTo(x2, y2)
    ctx.strokeStyle = "blue"
    ctx.stroke()
    ctx.closePath()

line(200,200,300,200)

line(200,200,200,300)

circle(200, 200, 10)
</script>
