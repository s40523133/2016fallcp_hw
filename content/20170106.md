<script type="text/python">
from browser import document
import math
 
# 準備繪圖畫布
canvas = document["dc1"]
ctx = canvas.getContext("2d")

def circle(x,y,r):
    ctx.beginPath()
    ctx.arc(x, y, r, 0, math.pi*2, True)
    ctx.fill()
    ctx.closePath()
ctx.beginPath()
# 設定線的寬度為 1 個單位
ctx.lineWidth = 1
# 將畫筆移動到 (200, 200) 座標點
ctx.moveTo(200, 200)
# 然後畫直線到 (150, 200) 座標點
ctx.lineTo(200, 300)
# 設定顏色為藍色, 也可以使用 "rgb(0, 0, 255)" 字串設定顏色值
ctx.strokeStyle = "blue"
# 實際執行畫線
ctx.stroke()
ctx.closePath()

circle(200, 200, 10)
</script>
